set(cxx-sources MessagableList.cpp)
set(ot-qobject-install-headers
    "${opentxs_SOURCE_DIR}/include/opentxs/ui/MessagableList.hpp"
)
set(cxx-install-headers ${ot-qobject-install-headers})
set(cxx-headers ${cxx-install-headers} MessagableList.hpp)

if(OT_WITH_QT)
  qt5_wrap_cpp(MOC_SRCS ${ot-qobject-install-headers})
else()
  set(MOC_SRCS "")
endif()

add_library(
  opentxs-ui-messagablelist OBJECT ${cxx-sources} ${cxx-headers} ${MOC_SRCS}
)
target_link_libraries(opentxs-ui-messagablelist PRIVATE opentxs::messages)

if(OT_QT_EXPORT)
  target_compile_definitions(
    opentxs-ui-messagablelist PRIVATE ${Qt5Core_DEFINITIONS}
  )
  target_compile_definitions(
    opentxs-ui-messagablelist PRIVATE ${Qt5Gui_DEFINITIONS}
  )
  target_compile_definitions(
    opentxs-ui-messagablelist PRIVATE ${Qt5Widgets_DEFINITIONS}
  )

  if(OT_QML_EXPORT)
    target_compile_definitions(
      opentxs-ui-messagablelist PRIVATE ${Qt5Qml_DEFINITIONS}
    )
    target_compile_definitions(
      opentxs-ui-messagablelist PRIVATE -DOT_WITH_QML=1
    )
  else()
    target_compile_definitions(
      opentxs-ui-messagablelist PRIVATE -DOT_WITH_QML=0
    )
  endif()
else()
  target_compile_definitions(opentxs-ui-messagablelist PRIVATE -DOT_WITH_QML=0)
endif()

add_dependencies(opentxs-ui-messagablelist generated_code)

if(OPENTXS_STANDALONE)
  install(
    FILES ${cxx-install-headers}
    DESTINATION include/opentxs/ui
    COMPONENT dev
  )
endif()
